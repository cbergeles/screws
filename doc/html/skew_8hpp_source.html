103  Rotation<NumType> exp(const NumType& theta = (NumType)1) const
104  {
105  if (theta == (NumType)0)
106  {
107  return Rotation<NumType>();
108  }
109  else
110  {
111  NumType mag = angle();
112 
113  if (fabs(mag) < FLT_EPSILON)
114  {
115  return Rotation<NumType>();
116  }
117  else
118  {
119  // Rodriguez formula
120  Rotation<NumType> rGen;
121  rGen._data = rGen._data +
122  (_data/mag)*sin(mag*theta) + (_data*_data)/(mag*mag)*((NumType)1.0 - cos(mag*theta));
123 
124  return rGen;
125  }
126  }
