170  const NumType& operator () (const unsigned int& i, const unsigned int& j) const
171  {
172  assert(i >= 0 && i < 4 && j >= 0 && j < 4);
173  if (i == 3 && j != 3)
174  {
175  return _zeroVal;
176  }
177  else if (i == 3 && j == 3)
178  {
179  return _oneVal;
180  }
181  else if (i < 3 && j < 3)
182  {
183  return _R(i, j);
184  }
185  else if (i < 3 && j == 3)
186  {
187  return _T(i);
188  }
189  else
190  {
191  screws::ScrewException s("Index out of bounds.", __FILE__, __FUNCTION__, __LINE__);
192  throw s;
193  }
194  }
