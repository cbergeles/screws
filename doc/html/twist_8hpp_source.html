65  Twist(const HomogeneousTransform<NumType>& HT)
66  {
67  Eigen::Matrix<NumType, 3, 3> Ainv;
68 
69  // Create the skew matrix from the rotation part of the homogeneous transform
70  _skew = Skew<NumType>(HT.rotation());
71 
72  // p. 414 from Sastry
73  Translation<NumType> trans = HT.translation();
74 
75  // Get the norm of the rotation
76  NumType skewNorm = _skew.norm();
77 
78  // Pure rotation
79  if ( fabs(trans.norm()) < 1e-10)
80  {
81  Ainv = Eigen::Matrix<NumType, 3, 3>::Zero();
82  }
83  // Pure translation
84  else if (fabs(skewNorm) < 1e-10)
85  {
86  Ainv = Eigen::Matrix<NumType, 3, 3>::Identity();
87  }
88  else
89  {
90  Ainv = Eigen::Matrix<double, 3, 3>::Identity() +
91  _skew._data/2 +
92  (2*sin(skewNorm) - skewNorm*(1 + cos(skewNorm)))/(2*skewNorm*skewNorm*sin(skewNorm))*_skew._data*_skew._data;
93 
94  }
95  _velocity = Translation<NumType>(
96  Ainv(0, 0)*trans(0) + Ainv(0, 1)*trans(1) + Ainv(0, 2)*trans(2),
97  Ainv(1, 0)*trans(0) + Ainv(1, 1)*trans(1) + Ainv(1, 2)*trans(2),
98  Ainv(2, 0)*trans(0) + Ainv(2, 1)*trans(1) + Ainv(2, 2)*trans(2)
99  );
100 
101  _zeroVal = (NumType)0;
102  }

206  HomogeneousTransform<NumType> exp(const NumType& theta = (NumType)1) const
207  {
208  // p. 413 Sastry
209  NumType omegaNorm = _skew.angle();
210 
211  if (fabs(omegaNorm) < 1e-10 || theta == (NumType)0) // pure translation or identity matrix
212  {
213  return HomogeneousTransform<NumType>(Rotation<NumType>(), _velocity*theta);
214  }
215  else
216  {
217  Eigen::Matrix<NumType, 3, 3> temp = _skew._data/(theta*omegaNorm*omegaNorm);
218  Eigen::Matrix<NumType, 3, 3> temp2 = temp*_skew._data/omegaNorm;
219 
220  Rotation<NumType> R = _skew.exp(theta);
221  Eigen::Matrix<NumType, 3, 3> A =
222  Eigen::Matrix<NumType, 3, 3>::Identity() +
223  temp*(1 - cos(omegaNorm*theta)) +
224  temp2*(omegaNorm*theta - sin(omegaNorm*theta));
225 
226  Eigen::Matrix<NumType, 3, 1, 0, 3, 1> v = A*_velocity._data*theta;
227 
228  return HomogeneousTransform<NumType>(R, Translation<NumType>(v(0), v(1), v(2)));
229  }
